CREATE TABLE customers
    (name CHAR(20),
    address CHAR(50),
    phone CHAR(10),
    PRIMARY KEY(phone))

CREATE TABLE orders
    (oid INTEGER,
    odate DATE,
    del_address CHAR(50),
    placed_by CHAR(10) NOT NULL,
    PRIMARY KEY(oid),
    FOREIGN KEY(placed_by)
      REFERENCES customers
      ON DELETE NO ACTION)


CREATE TABLE reviews
    (revId INTEGER,
    rdate DATE,
    rtext CHAR(500)
    oid INTEGER NOT NULL,
    PRIMARY KEY(revId),
    FOREIGN KEY (oid)
      REFERENCES orders
      ON DELETE NO ACTION)

CREATE TABLE deliveries
    (trackingNo INTEGER,
    plate CHAR(7) NOT NULL,
    PRIMARY KEY (trackingNo)
    FOREIGN KEY (plate)
      REFERENCES drivers,
      ON DELETE NO ACTION)

CREATE TABLE drivers
    (plate CHAR(7),
    name CHAR(20),
    phone CHAR(10),
    PRIMARY KEY(plate))

CREATE TABLE categories
    (cid INTEGER,
    name CHAR(20),
    PRIMARY KEY(cid))

CREATE TABLE products
    (upc INTEGER,
    name CHAR(20),
    unit CHAR(20),
    cid INTEGER NOT NULL,
    PRIMARY KEY(upc)
    FOREIGN KEY(cid)
      REFERENCES categories
      ON DELETE NO ACTION)

CREATE TABLE stores
    (sid INTEGER,
    name CHAR(20),
    phone CHAR(10),
    address CHAR(50),
    PRIMARY KEY(sid))

CREATE TABLE  lines
    (lno INTEGER,
    qty INTEGER,
    uprice FLOAT,
    oid INTEGER NOT NULL,
    sid INTEGER NOT NULL,
    upc INTEGER NOT NULL,
    PRIMARY KEY (lno,oid),
    FOREIGN KEY (sid)
      REFERENCES stores
      ON DELETE NO ACTION,
    FOREIGN KEY (upc)
      REFERENCES products
      ON DELETE NO ACTION,
    FOREIGN KEY (oid)
      REFERENCES orders
      ON DELETE CASCADE)

CREATE TABLE carries(
    sid INTEGER,
    upc INTEGER,
    qty INTEGER,
    price FLOAT,
    PRIMARY KEY (upc, sid),
    FOREIGN KEY (upc)
      REFERENCES products,
    FOREIGN KEY (sid)
      REFERENCES stores)

CREATE TABLE incl(
    oid INTEGER,
    trackingNo INTEGER NOT NULL,
    pickupTime CHAR(10),
    dropOffTime CHAR(10),
    PRIMARY KEY (trackingNo, oid),
    FOREIGN KEY (oid)
      REFERENCES orders,
    FOREIGN KEY (trackingNo)
      REFERENCES deliveries)
